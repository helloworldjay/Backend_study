-- select 구문

select * from departments;

select deptno, name from department;

select deptno 부서번호, name 부서명 from department; (Alias 설정)

- 컬럼의 합성(Concatenation)

문자열 결합함수 concat을 사용한다

ex) employee 테이블에서 사번과 부서번호를 하나의 칼럼으로 출력

SELECT concat(empno, '-', deptno) as '사번-부서번호' FROM employee;

- 중복제거

SELECT DISTINCT deptno FROM employee;

- 특정 칼럼을 기준으로 정렬
> select empno, name from employee order by name; (칼럼명으로 정렬)

> select empno, name from employee order by 1; (1번째 칼럼 순서로 정렬)

> select empno, name from employee order by name desc; (내림차순 정렬)

- WHERE절 사용

사원 번호가 10번이거나 30번인 사람만 찾기

select * from employee where deptno in (10, 30);

- LIKE 연산

와일드 카드 % _ 

이름이 A로 시작하는 사람 찾기

SELECT * FROM employee where name LIKE 'A%';

- 대문자 소문자 함수

대문자 : SELECT UPPER('seoul'); or SELECT UCASE('seoul');

소문자 : SELECT LOWER('SEoul'); or SELECT LCASE('SEoul');

 - substring

SELECT SUBSTRING('Happy Day', 3, 2);    mysql 은 index가 0이 아닌 1부터 시작한다. 3,2 는 3번 인덱스부터 시작해서 2글자 출력을 의미

-LPAD RPAD (내가 지정한 문자를 채워줘)

SELECT LPAD('hi', 5, '?'), LPAD('joe',7,'*'); hi의 왼쪽에 ?를 채워 총 5칸으로 만들어줘

문자열 자리에 column을 넣어 출력할 수 도있다.

- TRIM, LTRIM, RTRIM 공백 제거

SELECT LTRIM('HELLO ');

- CAST 형변환

CAST(expression AS type) 또는 CONVERT(expression, type)
